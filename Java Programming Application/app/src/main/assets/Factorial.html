import java.util.Scanner;

class Factorial

{

public static void main(String args[])

{

int n, c, fact = 1;

System.out.println("Enter number to calculate factorial");

Scanner in = new Scanner(System.in);

n = in.nextInt();

for ( c = 1 ; c <= n ; c++ )

fact = fact*c;

System.out.println("Factorial "+n+" is = "+fact);

}

}
